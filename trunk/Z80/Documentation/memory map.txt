0x0040		disk sector
0x0044		busy flag (disk operations)
0x0045		cursor column
0x0046		cursor row
0x0050		controller command
0x0051		controller param1
0x0053		controller param2
0x0055		controller outparam
0x0058		program parameter count (max. 8)
0x0060-0x007f	sparam (32 bytes)
0x0080-0x027f	disk r/w buffer (512 bytes), program parameters (8 x 64 bytes)
0x0280		start of program code
0x1000-0x1fff	video ram (4K bytes)
